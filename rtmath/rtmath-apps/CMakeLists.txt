# Build the apps

macro(addappdir appname appdefault)
	option(BUILD_APP_${appname}
		"Build ${appname} app"
		${appdefault})
	if(BUILD_APP_${appname})
		add_subdirectory(${appname})
	endif()
endmacro(addappdir appname appdefault)
